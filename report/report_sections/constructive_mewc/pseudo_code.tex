\begin{algorithm}[H]
    \caption{constructiveMEWC function}
    \label{constructivemewc-pseudocode}
    \begin{algorithmic}[1]
        \Procedure{constructiveMEWC}{$G$}
        \State $S \gets \emptyset$ \Comment{Variable to store the clique that represents the Solution}
        \State $P \gets G.\text{vertices()}$ \Comment{Lists of vertex to store the vertex that could be added to $S$}
        \State $Z \gets \Call{sortVerticesDegree}{G, P}$ \Comment{variable to store the vertices sorted by degree}
        \State \Call{ConstructiveRecursiveMEWC}{$G$, $S$, $P$, $Z$}
        \State \Call{getCliqueWeight}{$S$}
        \State \Return S
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
    \caption{constructiveMEWCRecursive function}
    \label{constructivemewcrecursive-pseudocode}
    \begin{algorithmic}[1]
        \Procedure{constructiveMEWCRecursive}{$G$, $S$, $P$, $Z$}
        \If {$P = \emptyset$}
        \State \Return
        \EndIf
        \State $v \gets \Call{getBestVertex}{P, Z}$ \Comment{Get the vertex with the highest degree}
        \State $S \gets S \cup \{v\}$
        \State \Call{ConstructiveRecursiveMEWC}{$G$, $S$, $P \cap (N(v) \cup {v})$, $Z$}
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
    \caption{getBestVertex function}
    \label{getbestvertex-pseudocode}
    \begin{algorithmic}[1]
        \Procedure{getBestVertex}{$P$, $Z$}
        \State $bestVertex \gets \emptyset$
        \For{$v \in Z$}
        \If {$\exists v \in P$}
        \State $bestVertex \gets v$
        \State $Z \gets Z - \{v\}$
        \State \textbf{break}
        \EndIf
        \EndFor
        \State \Return $bestVertex$
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
    \caption{sortVerticesDegree function}
    \label{sortverticesdegree-pseudocode}
    \begin{algorithmic}[1]
        \Procedure{sortVerticesDegree}{$G$, $P$}
        \State $degrees \gets (\emptyset, \emptyset)$ \Comment{Make a pair (vertex, degree)}
        \State $Z \gets \emptyset$

        \State $adjMatrix \gets G.getAdjacencyMatrix()$

        \ForAll{$vertex \in P$}
        \State $neighbors \gets adjMatrix[vertex.id()]$
        \State $degrees.push\_back(vertex, |N|)$
        \EndFor
        \State \Call{sort}{degrees} \Comment{Sort the vertices by degree}
        \ForAll{$[vertex, degree] \in degrees$}
        \State $Z.push\_back(vertex)$
        \EndFor
        \State \Return $Z$
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
    \caption{Clique weight function}
    \label{clique-weight2-pseudocode}
    \begin{algorithmic}[1]
        \Procedure{getCliqueWeight}{$S$} \Comment{We assume that the clique is complete}
        \State $weight \gets 0$
        \State $n \gets |S|$
        \ForAll{$i \in \{1,2,\ldots,n\}$} \Comment{For each vertex in the clique}
        \ForAll{$j \in \{i+1,i+2,\ldots,n\}$} \Comment{For each other vertex in the clique}
        \State $weight \gets weight + w_{ij}$ \Comment{Add the weight of the edge between the two vertices}
        \EndFor
        \EndFor
        \State \Return $weight$
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
